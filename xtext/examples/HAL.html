Automation rule

Name Rule
Simple WHEN "temperature sensor #1" IN "Living Room" < "18"
THEN
heater.turnOffHeating IN "Living Room"
Compound WHEN ("temperature sensor #1" IN "Living Room" < "18" AND window IN "Living Room" = "closed")
THEN
heater.turnOnHeating IN "Living Room"
Nested WHEN (("temperature sensor #1" IN "Living Room" < "18" OR NOT ("temperature sensor #2" IN "Living Room" >= "18")) AND window IN "Living Room" = "closed")
THEN
heater.turnOnHeating IN "Living Room"
heater.turnOnHeating IN Kitchen
Complex WHEN (((("temperature sensor #1" IN "Living Room" > "24" AND "temperature sensor #1" IN Kitchen > "24") OR "temperature sensor #1" IN "Living Room" > "26") OR "temperature sensor #1" IN Kitchen > "26") AND (window IN "Living Room" = "closed" OR window IN Kitchen = "closed"))
THEN
heater.turnOffHeating IN "Living Room"
autoGlass.open IN "Living Room"
heater.turnOffHeating IN Kitchen
smartShard.open IN Kitchen
